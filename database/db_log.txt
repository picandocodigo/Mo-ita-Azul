CREATE TABLE `monita`.`ma_balance` (
  `id` INT  NOT NULL AUTO_INCREMENT,
  `name` VARCHAR(100)  NOT NULL,
  `year` INT(4)  NOT NULL,
  `credit` BIGINT(15)  NOT NULL,
  `expenses` BIGINT(15)  NOT NULL,
  PRIMARY KEY (`id`)
)
ENGINE = InnoDB
CHARACTER SET utf8 COLLATE utf8_general_ci;
